import pandas as pd
import numpy as np
from sklearn.tree import DecisionTreeRegressor
from sklearn.metrics import r2_score

# 예제 데이터 생성
data = {'weight': [60, 70, 80, 90],
        'sex': ['male', 'female', 'male', 'female'],
        'y': [100, 150, 200, 250]}
df = pd.DataFrame(data)

# (3)에서 구한 예측값 yhat
yhat = np.array([110, 160, 210, 260])

# (6)에서 구한 잔차 r
r = np.array([90, -10, -10, -10])

# (5)에서 구한 예측값 rhat (의사결정나무에 의한 예측)
X = pd.get_dummies(df[['weight', 'sex']])
predictr = DecisionTreeRegressor(max_depth=2)
predictr.fit(X, r)
rhat = predictr.predict(X)

# 학습률 설정
learning_rate = 0.1

# (6)과 (3)-(5)의 결과를 이용하여 yhat2 및 r2 계산
yhat2 = yhat + learning_rate * rhat
r2 = r - learning_rate * rhat

# (3)-(5)에서 구한 예측값 yhat을 업데이트
df['yhat2'] = yhat2

# (3)-(5)에서 구한 r2_score 계산
r2_score_result = r2_score(df['y'], df['yhat2'])

# 결과 출력
print("Updated yhat2:")
print(df[['weight', 'sex', 'yhat2']])
print("\nR-squared (r2_score) between y and yhat2:", r2_score_result)

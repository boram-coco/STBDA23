        lw=3
    )

    # step1 -- pass 
    # step2

    # step3
    predictr.fit(df_train_featured)
    # step4 
    yhat = predictr.predict(df_train_featured)

    # display
    display(predictr.leaderboard())
    return yhat, yyhat
